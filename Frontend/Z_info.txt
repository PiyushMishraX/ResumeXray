The React Context API setup involves three core steps: creating the context, providing the state, and consuming the value. 1.  Create Context: Use createContext() to define a context object.  2. Create Provider: Build a component that holds the state (using useState or useReducer) and renders <Context.Provider value={...}> to wrap child components. 3.  Consume Context: Use the useContext() hook in any descendant component to access the provided data. 

import React, { createContext, useState, useContext } from 'react';

// 1. Create the context
const ThemeContext = createContext();

// 2. Create the Provider component
export const ThemeProvider = ({ children }) => {
  const [theme, setTheme] = useState('light');
  
  const toggleTheme = () => {
    setTheme(prev => prev === 'light' ? 'dark' : 'light');
  };

  return (
    <ThemeContext.Provider value={{ theme, toggleTheme }}>
      {children}
    </ThemeContext.Provider>
  );
};

// Export the context for consumption
export default ThemeContext;
